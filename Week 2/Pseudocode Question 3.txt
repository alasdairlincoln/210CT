matrixAddition(B,C)
	for i in B
		for j in i
			d[i] <- j

	for i in C
		for j in i
			d[i] <- d[i] + j

return(d)

matrixSubtraction(B,C)
	for i in B
		for j in i
			d[i] <- j

	for i in C
		for j in i
			d[i] <- d[i] - j

return(d)

matrixMultiplication(B,C)
	for i in B
		for j in i
			d[i] <- j

	for i in C
		for j in i
			d[i] <- d[i] * j
